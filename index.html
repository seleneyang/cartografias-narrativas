function mulberry32(a) {

  return function() {

    let t = (a += 0x6D2B79F5);

    t = Math.imul(t ^ (t >>> 15), t | 1);

    t ^= t + Math.imul(t ^ (t >>> 7), t | 61);

    return ((t ^ (t >>> 14)) >>> 0) / 4294967296;

  };

}

function setMarker(lat, lng) {

  }

  updateCardMap();

}

async function geocode() {

  const q = document.getElementById('address').value.trim();

  if (!q) return;

  const btn = document.getElementById('searchBtn');

  btn.disabled = true;

  btn.textContent = '…';

  try {

    const url = `https://nominatim.openstreetmap.org/search?q=${encodeURIComponent(q)}&format=json&limit=1`;

    const res = await fetch(url, { headers: { 'Accept': 'application/json' } });

    const data = await res.json();

    if (data && data[0]) {

      const lat = parseFloat(data[0].lat);

      const lng = parseFloat(data[0].lon);

      editorMap.setView([lat, lng], 14);

      setMarker(lat, lng);

    } else {

      alert('No se encontró esa dirección. Probá con una variante o hacé click en el mapa.');

    }

  } catch (e) {

    alert('El servicio de búsqueda no respondió. Podés ubicar el pin manualmente.');

  } finally {

    btn.disabled = false;

    btn.textContent = 'Buscar';

  }

}

function getIntensities() {

  return EMOTIONS.map(e => ({

    ...e,

    value: parseInt(document.getElementById('emo-' + e.key).value, 10)

  }));

}


function renderBlob() {

  const svg = document.getElementById('blob');

  svg.innerHTML = '';

  const rng = mulberry32(seed);

  const cx = 100, cy = 100;

  const intensities = getIntensities();

  const total = intensities.reduce((s, e) => s + e.value, 0);

  if (total === 0) return;

  const pool = [];

  intensities.forEach(e => {

    for (let i = 0; i < e.value * 3; i++) pool.push(e.color);

  });

  for (let i = pool.length - 1; i > 0; i--) {

    const j = Math.floor(rng() * (i + 1));

    [pool[i], pool[j]] = [pool[j], pool[i]];

  }

  const g = document.createElementNS('http://www.w3.org/2000/svg', 'g');

  g.setAttribute('style', 'mix-blend-mode: multiply');

  svg.appendChild(g);

  pool.forEach(color => {

    const u1 = Math.max(1e-9, rng()), u2 = rng();

    const z0 = Math.sqrt(-2 * Math.log(u1)) * Math.cos(2 * Math.PI * u2);

    const z1 = Math.sqrt(-2 * Math.log(u1)) * Math.sin(2 * Math.PI * u2);

    const x = cx + z0 * 32;

    const y = cy + z1 * 32;

    const r = 10 + rng() * 16;

    const opacity = 0.32 + rng() * 0.28;

    const c = document.createElementNS('http://www.w3.org/2000/svg', 'circle');

    c.setAttribute('cx', x.toFixed(2));

    c.setAttribute('cy', y.toFixed(2));

    c.setAttribute('r', r.toFixed(2));

    c.setAttribute('fill', color);

    c.setAttribute('opacity', opacity.toFixed(2));

    g.appendChild(c);

  });

}


function update() {

  EMOTIONS.forEach(e => {

    const v = document.getElementById('emo-' + e.key).value;

    document.getElementById('val-' + e.key).textContent = v;

  });

  document.getElementById('cardPlace').textContent =

    document.getElementById('place').value || 'Un lugar';

  document.getElementById('cardAuthor').textContent =

    document.getElementById('author').value || 'Anónimo';

  renderBlob();

  if (editorMarker) editorMarker.setIcon(makePin(dominantColor()));

  if (cardMarker) cardMarker.setIcon(makePin(dominantColor()));

}


function regenerate() {

  seed = Math.floor(Math.random() * 1e9);

  renderBlob();

}


function slug(s) {

  return (s || 'lugar').toLowerCase()

    .normalize('NFD').replace(/[\u0300-\u036f]/g, '')

    .replace(/[^a-z0-9]+/g, '-').replace(/^-|-$/g, '');

}


function downloadSVG() {

  const inner = document.getElementById('blob').innerHTML;

  const full = `<?xml version="1.0" encoding="UTF-8"?>

<svg xmlns="http://www.w3.org/2000/svg" viewBox="0 0 200 200" width="400" height="400">

${inner}

</svg>`;

  const blob = new Blob([full], { type: 'image/svg+xml' });

  const url = URL.createObjectURL(blob);

  const a = document.createElement('a');

  a.href = url;

  a.download = `tarjeta-${slug(document.getElementById('place').value)}.svg`;

  a.click();

  URL.revokeObjectURL(url);

}


function exportJSON() {

  const data = {

    author: document.getElementById('author').value,

    place: document.getElementById('place').value,

    address: document.getElementById('address').value,

    lat: Number(currentLat.toFixed(6)),

    lng: Number(currentLng.toFixed(6)),

    seed,

    emotions: getIntensities().reduce((acc, e) => {

      acc[e.key] = e.value;

      return acc;

    }, {}),

    generated_at: new Date().toISOString()

  };

  const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' });

  const url = URL.createObjectURL(blob);

  const a = document.createElement('a');

  a.href = url;

  a.download = `tarjeta-${slug(document.getElementById('place').value)}.json`;

  a.click();

  URL.revokeObjectURL(url);

}


function buildPayload() {

  return {

    author: document.getElementById('author').value,

    place: document.getElementById('place').value,

    address: document.getElementById('address').value,

    lat: Number(currentLat.toFixed(6)),

    lng: Number(currentLng.toFixed(6)),

    seed,

    emotions: getIntensities().reduce((acc, e) => {

      acc[e.key] = e.value;

      return acc;

    }, {}),

    generated_at: new Date().toISOString()

  };

}


async function sendToGallery() {

  if (!ENDPOINT) return;

  const btn = document.getElementById('sendBtn');

  const orig = btn.textContent;

  btn.disabled = true;

  btn.textContent = 'Enviando…';

  try {

    // Usamos text/plain para evitar preflight CORS en Apps Script

    const res = await fetch(ENDPOINT, {

      method: 'POST',

      headers: { 'Content-Type': 'text/plain;charset=utf-8' },

      body: JSON.stringify(buildPayload())

    });

    const result = await res.json();

    if (result && result.ok) {

      btn.textContent = '¡Enviada! ✓';

      setTimeout(() => { btn.textContent = orig; btn.disabled = false; }, 2200);

    } else {

      throw new Error((result && result.error) || 'error');

    }

  } catch (e) {

    console.error(e);

    btn.textContent = 'Falló · reintentar';

    setTimeout(() => { btn.textContent = orig; btn.disabled = false; }, 2500);

  }

}
