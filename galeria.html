const ENDPOINT = 'https://script.google.com/macros/s/AKfycbxaKMpyp9YmTdvpmn-LI7beUCPsPUbP_Icth5fk_m-TSSph_fSIReRCIRQkyx6njL72/exec';


// Paleta — debe coincidir con la del generador

const COLORS = {

  alegria:    '#F4B942',

  calidez:    '#E76F51',

  amor:       '#E63946',

  nostalgia:  '#B298DC',

  melancolia: '#457B9D',

  paz:        '#7FC4B2',

  naturaleza: '#6BAA75',

  sagrado:    '#6A4C93',

};


function mulberry32(a) {

  return function() {

    let t = (a += 0x6D2B79F5);

    t = Math.imul(t ^ (t >>> 15), t | 1);

    t ^= t + Math.imul(t ^ (t >>> 7), t | 61);

    return ((t ^ (t >>> 14)) >>> 0) / 4294967296;

  };

}


function dominantColor(emotions) {

  let max = 0, key = 'calidez';

  for (const [k, v] of Object.entries(emotions)) {

    if (v > max) { max = v; key = k; }

  }

  return COLORS[key] || '#e76f51';

}


function renderBlob(carta) {

  const cx = 100, cy = 100;

  const rng = mulberry32(carta.seed);

  const pool = [];

  for (const [key, value] of Object.entries(carta.emotions)) {

    const color = COLORS[key];

    if (!color) continue;

    for (let i = 0; i < value * 3; i++) pool.push(color);

  }

  if (pool.length === 0) return '<svg class="blob" viewBox="0 0 200 200"></svg>';


  for (let i = pool.length - 1; i > 0; i--) {

    const j = Math.floor(rng() * (i + 1));

    [pool[i], pool[j]] = [pool[j], pool[i]];

  }


  const circles = pool.map(color => {

    const u1 = Math.max(1e-9, rng()), u2 = rng();

    const z0 = Math.sqrt(-2 * Math.log(u1)) * Math.cos(2 * Math.PI * u2);

    const z1 = Math.sqrt(-2 * Math.log(u1)) * Math.sin(2 * Math.PI * u2);

    const x = cx + z0 * 32;

    const y = cy + z1 * 32;

    const r = 10 + rng() * 16;

    const op = 0.32 + rng() * 0.28;

    return `<circle cx="${x.toFixed(2)}" cy="${y.toFixed(2)}" r="${r.toFixed(2)}" fill="${color}" opacity="${op.toFixed(2)}"/>`;

  }).join('');


  return `<svg class="blob" viewBox="0 0 200 200" xmlns="http://www.w3.org/2000/svg">

    <g style="mix-blend-mode: multiply">${circles}</g>

  </svg>`;

}


function escapeHTML(s) {

  return String(s || '').replace(/[&<>"']/g, m => ({

    '&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'

  }[m]));

}


function slug(s) {

  return (s || 'lugar').toLowerCase()

    .normalize('NFD').replace(/[\u0300-\u036f]/g, '')

    .replace(/[^a-z0-9]+/g, '-').replace(/^-|-$/g, '');

}


function makePin(color, big = false) {

  return L.divIcon({

    className: '',

    html: `<div class="custom-pin ${big ? 'big' : ''}" style="background:${color}"></div>`,

    iconSize: big ? [28, 28] : [20, 20],

    iconAnchor: big ? [14, 14] : [10, 10],

  });

}


function renderGallery(cartas) {

  const grid = document.getElementById('grid');

  if (!cartas.length) {

    grid.innerHTML = '<div class="status">Todavía no hay tarjetas. Generá la primera con <code>index.html</code>.</div>';

    return;

  }


  // HTML de las tarjetas con un div vacío para el mapa

  grid.innerHTML = cartas.map((c, i) => {

    const id = `card-${i}-${slug(c.place)}`;

    const hasCoords = typeof c.lat === 'number' && typeof c.lng === 'number';

    return `

      <article class="card" id="${id}">

        <div class="card-row">

          ${renderBlob(c)}

          <div>

            <p class="place">${escapeHTML(c.place)}</p>

            <div class="author">${escapeHTML(c.author)}</div>

          </div>

        </div>

        ${hasCoords ? `<div class="card-map" id="miniMap-${i}"></div>` : ''}

      </article>

    `;

  }).join('');


  // Instanciar los mini-mapas después que el DOM esté listo

  cartas.forEach((c, i) => {

    if (typeof c.lat !== 'number' || typeof c.lng !== 'number') return;

    const el = document.getElementById(`miniMap-${i}`);

    if (!el) return;

    const m = L.map(el, {

      zoomControl: false,

      dragging: false,

      scrollWheelZoom: false,

      doubleClickZoom: false,

      boxZoom: false,

      keyboard: false,

      touchZoom: false,

      attributionControl: false,

    }).setView([c.lat, c.lng], 13);

    L.tileLayer('https://tile.openstreetmap.org/{z}/{x}/{y}.png', {

      maxZoom: 18,

    }).addTo(m);

    L.marker([c.lat, c.lng], { icon: makePin(dominantColor(c.emotions)) }).addTo(m);

    setTimeout(() => m.invalidateSize(), 200 + i * 50);

  });

}


function renderBigMap(cartas) {

  const withCoords = cartas.filter(c => typeof c.lat === 'number' && typeof c.lng === 'number');

  if (!withCoords.length) {

    document.getElementById('bigMap').innerHTML =

      '<div class="status">Aún no hay tarjetas con ubicación.</div>';

    return;

  }


  const bigMap = L.map('bigMap', {

    scrollWheelZoom: false,

  }).setView([withCoords[0].lat, withCoords[0].lng], 4);


  L.tileLayer('https://tile.openstreetmap.org/{z}/{x}/{y}.png', {

    maxZoom: 18,

    attribution: '&copy; OpenStreetMap'

  }).addTo(bigMap);


  const markers = [];

  withCoords.forEach((c, i) => {

    const color = dominantColor(c.emotions);

    const m = L.marker([c.lat, c.lng], { icon: makePin(color, true) }).addTo(bigMap);

    const id = `card-${cartas.indexOf(c)}-${slug(c.place)}`;

    m.bindPopup(`

      <p class="popup-place">${escapeHTML(c.place)}</p>

      <div class="popup-author">${escapeHTML(c.author)}</div>

      <a class="popup-link" data-target="${id}">ver la tarjeta ↑</a>

    `);

    markers.push(m);

  });


  // Ajustar el zoom para que entren todos los puntos

  if (withCoords.length > 1) {

    const group = L.featureGroup(markers);

    bigMap.fitBounds(group.getBounds().pad(0.3));

  } else {

    bigMap.setView([withCoords[0].lat, withCoords[0].lng], 13);

  }


  // Click en "ver la tarjeta" del popup = scroll hasta ella

  bigMap.on('popupopen', e => {

    const link = e.popup._contentNode.querySelector('.popup-link');

    if (!link) return;

    link.addEventListener('click', ev => {

      ev.preventDefault();

      const target = document.getElementById(link.dataset.target);

      if (!target) return;

      target.scrollIntoView({ behavior: 'smooth', block: 'center' });

      target.classList.add('highlight');

      setTimeout(() => target.classList.remove('highlight'), 2500);

    });

  });


  setTimeout(() => bigMap.invalidateSize(), 300);

}

async function loadCards() {

  if (ENDPOINT) {

    try {

      const res = await fetch(ENDPOINT);

      if (res.ok) {

        const data = await res.json();

        if (Array.isArray(data) && data.length) {

          renderGallery(data);

          renderBigMap(data);

          return;

        }

      }

    } catch (e) {

      console.warn('Endpoint no disponible, probando cartas.json', e);

    }

  }

  try {

    const res = await fetch('cartas.json');

    if (!res.ok) throw new Error('no json');

    const data = await res.json();

    renderGallery(data);

    renderBigMap(data);

  } catch (e) {

    renderGallery(EJEMPLOS);

    renderBigMap(EJEMPLOS);

  }

}
